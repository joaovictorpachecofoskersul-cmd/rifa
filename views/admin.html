// let usuarioId = null; // let rifaId = null; // Pegar parâmetros da URL const urlParams = new URLSearchParams(window.location.search); const usuarioId = urlParams.get('usuarioId'); const rifaId = urlParams.get('rifaId'); if (!usuarioId || !rifaId) { alert('Parâmetros inválidos! Volte ao dashboard.'); window.location.href = '/dashboard'; } function fazerLogin() { const email = document.getElementById('loginEmail').value; const senha = document.getElementById('senhaAdmin').value; if (!email || !senha) { document.getElementById('errorMsg').textContent = 'Preencha email e senha!'; document.getElementById('errorMsg').style.display = 'block'; return; } fetch('/api/login', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ email, senha }) }) .then(res => res.json()) .then(data => { if (data.success) { // Verificar se o usuário logado é o dono da rifa if (data.usuario.id !== usuarioId) { document.getElementById('errorMsg').textContent = 'Você não tem permissão para acessar esta rifa!'; document.getElementById('errorMsg').style.display = 'block'; return; } localStorage.setItem('adminLogado', 'true'); localStorage.setItem('usuarioId', data.usuario.id); document.getElementById('loginOverlay').style.display = 'none'; document.getElementById('mainWrapper').style.display = 'block'; carregarDashboard(); carregarVendasPendentes(); carregarConfiguracoes(); } else { document.getElementById('errorMsg').textContent = data.error; document.getElementById('errorMsg').style.display = 'block'; setTimeout(() => { document.getElementById('errorMsg').style.display = 'none'; }, 3000); } }); } function fazerLogout() { localStorage.removeItem('adminLogado'); localStorage.removeItem('usuarioId'); document.getElementById('loginOverlay').style.display = 'flex'; document.getElementById('mainWrapper').style.display = 'none'; } function getHeaders() { return { 'Content-Type': 'application/json', 'user-token': localStorage.getItem('usuarioId') }; }
